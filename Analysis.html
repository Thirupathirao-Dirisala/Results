packages = ["pandas","matplotlib"] import pandas as pd import js import matplotlib.pyplot as plt r=js.localStorage.getItem('myValue') y=int(js.localStorage.getItem('myYear')) s=int(js.localStorage.getItem('mySem')) from pyodide.http import open_url for k in range(1,y+1): for j in range(1,3): if k==y and j>=s: break else: l1=[] l2=[] l3=[] df = pd.read_csv(open_url(f'https://raw.githubusercontent.com/Thirupathirao-Dirisala/Results/main/SEM-{k}.{j}.csv')) x=df.loc[df["REDG NO"] == r] d = x.to_dict(orient='records') l1 = list(d[0].keys())[4:] l2 = list(d[0].values())[4:] for i in l2: l3.append(int(i.split()[0])) display(f'SEM-{k}.{j} Results') b=plt.bar(l1,l3,facecolor="green",label=l3) display(b) p=plt.pie(l3,labels=l1,autopct="%.0f%%") display(p)
